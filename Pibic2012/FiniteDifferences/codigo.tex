\section{Equaçoes}

\NamedEquation{
 {
u_{i,j} = \frac{1}{4}(u_{i+1,j} + u_{i-1,j} + u_{i,j+1} + u_{i,j-1} - f_{i,j}\triangle x ^2) }

\NamedEquation
 {
u_{i,j} = \frac{2}{3}(2u_{i,j+1} - \frac{1}{2} u_{i,j+2} - \triangle xf_{i,j})

\NamedEquation
 {
u_{i,j} = \frac{2}{3}(2u_{i,j-1} - \frac{1}{2} u_{i,j-2} + \triangle xf_{i,j})

\NamedEquation
 {
u_{i,j} = \frac{2}{3}(2u_{i-1,j} - \frac{1}{2} u_{i-2,j} + \triangle xf_{i,j})
}{borda topo}

\NamedEquation
 {
u_{i,j} = \frac{2}{3}(2u_{i+1,j} - \frac{1}{2} u_{i+2,j}  \triangle xf_{i,j})
}{borda fundo}
